bool isanagram(string s , string t){
      if(s.length() != t.length()){
           return false ;
       }
     vector<int > freq(26 , 0){
      for(int i =0;i<s.length();i++){
           freq[s[i]-'a']++;
           freq[t[i]-'a']--;
        }
       for(int count : freq){
           if(count != 0){
              return false ;
           } 
          }
       return true;
     }
}
2 . first unique character in string :-
class Solution {
public:
    int firstUniqChar(string s) {
        int n = s.length();
        vector<int> freq(26 ,0);
        for(char ch : s){
            freq[ch -'a']++;
        }
        for(int i = 0;i<n;i++){
            if(freq[s[i]-'a']==1){
                return i;
            }
        }
        return -1;
    }
};
3 . ransome note :-
class Solution {
public:
    bool canConstruct(string ransomNote, string magazine) {
        vector<int> freq(26 ,0);
        for(char ch : magazine){
            freq[ch - 'a']++;
        }
        for(char ch : ransomNote){
            freq[ch -'a']--;
            if(freq[ch - 'a']< 0){
                return false ;
            }
        }
        return true;
    }
};
