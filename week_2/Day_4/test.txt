class Solution {
public:
    string reverseWords(string s) {

        int start = 0;

        for(int end = 0; end <= s.size(); end++)
        {
            if(end == s.size() || s[end] == ' ')
            {
                reverse(s.begin() + start, s.begin() + end);

                start = end + 1;
            }
        }

        return s;
    }
};
2. DECODE STRING :-
class Solution {
public:
    string decodeString(string s) {

        stack<int> countStack;
        stack<string> stringStack;

        string currString = "";
        int currNum = 0;

        for(char ch : s)
        {
            if(isdigit(ch))
            {
                currNum = currNum * 10 + (ch - '0');
            }
            else if(ch == '[')
            {
                countStack.push(currNum);
                stringStack.push(currString);

                currNum = 0;
                currString = "";
            }
            else if(ch == ']')
            {
                int repeat = countStack.top();
                countStack.pop();

                string prevString = stringStack.top();
                stringStack.pop();

                string temp = "";

                for(int i = 0; i < repeat; i++)
                {
                    temp += currString;
                }

                currString = prevString + temp;
            }
            else
            {
                currString += ch;
            }
        }

        return currString;
    }
};
3. GENERTAE PARANTHESES :-
class Solution {
public:

    void solve(int open, int close, string curr, vector<string>& ans, int n)
    {
        if(curr.length() == 2 * n)
        {
            ans.push_back(curr);
            return;
        }

        // Add '(' if we still can
        if(open < n)
        {
            solve(open + 1, close, curr + '(', ans, n);
        }

        // Add ')' only if it won't make the string invalid
        if(close < open)
        {
            solve(open, close + 1, curr + ')', ans, n);
        }
    }

    vector<string> generateParenthesis(int n)
    {
        vector<string> ans;

        solve(0, 0, "", ans, n);

        return ans;
    }
};
